	\section{$\lambda$-calculs }
	
	
	\begin{frame}
		\subsection{Bases}
		\frametitle{$\lambda$-calculs }
		\begin{block}{Définition}
			Le $\lambda$-calcul est un système formel inventé par Alonzo Church dans les années 1930, qui fonde les concepts de fonction et
			d'application.
		\end{block}
	
		\begin{columns}
			\begin{column}{3cm}
				\begin{semiverbatim}
					Variable
					\smallbreak 
					$X$
				\end{semiverbatim}	
			
			\end{column}
			
			\begin{column}{3cm}
				\begin{semiverbatim}
				Abstraction
				\smallbreak 
				$\lambda X.M$
			\end{semiverbatim}	
			\end{column}
			
			\begin{column}{3cm}
				\begin{semiverbatim}
				Application
				\smallbreak 
				$(M~N)$
			\end{semiverbatim}	
			\end{column}
		\end{columns}
		
				\begin{block}{Règles de réduction}
					\begin{itemize}
						\item[$\alpha$] $(\lambda X_{1}.M)~~~~~~~~~~\rightarrow_{\alpha}~~~(\lambda X_{2}.M[X_{1} \leftarrow X_{2}])$ où $X_{2}~\notin~FV(M)$ 
						\item[$\beta$] $((\lambda X.M_{1})M_{2})~~~\rightarrow_{\beta}~~~M_{1}[X \leftarrow M_{2}]$ 
						\item[$\eta$] $(\lambda X.(M~X))~~~~~\rightarrow_{\eta}~~~M$ où $X~\notin~FV(M)$ 
					\end{itemize}
				\end{block}
		
	
	\end{frame}

	
	\begin{frame}
		\subsection{Forme normale}
		\frametitle{$\lambda$-calculs }
		
		\begin{block}{Théorème de la forme normale}
			Si on peut réduire $L$ tels que $L =_{n}~M$ et $L =_{n}~N$ et que $N$ et $M$ sont en forme normale alors $M = N$ à n renommages près.
		\end{block}
		
		\begin{block}{Règles de la stratégie de réduction}
			\begin{itemize}
				\item[] $M~~~~~~~~~~~\longrightarrow_{\bar{n}}~~~N$~~~si $M~\beta~N$
				\item[] $M~~~~~~~~~~~\longrightarrow_{\bar{n}}~~~N$~~~si $M~\eta~N$
				\item[] $(\lambda X.M)~~~\longrightarrow_{\bar{n}}~~~(\lambda X.N)$ si on a $M~\beta~N$ ou $M~\eta~N$
				\item[] $(M~N)~~~~~\longrightarrow_{\bar{n}}~~~(M'~N)$ si $M \longrightarrow_{\bar{n}}~M'$ et $\forall~L$, $(M~N)~\cancel{\beta}~L$
				\item[] $(M~N)~~~~~\longrightarrow_{\bar{n}}~~~(M~N')$  si $N \longrightarrow_{\bar{n}}~N'$ et $\forall~L$, $(M~N)~\cancel{\beta}~L$
			\end{itemize}
		\end{block}

	\end{frame}

	\begin{frame}
		\frametitle{Machines étudiées}
		
		\subsection{Machine SECD}
		\textbf{Machine SECD}
		\begin{columns}
			\begin{column}{5cm}
				\begin{itemize}
				\item Sauvegarde différente du CEK 
				\item Appel par valeur
				\item Fonctionne avec son propre langage
			\end{itemize}
			\end{column}
			\begin{column}{5cm}
				\begin{itemize}
				\item Composée de quatres éléments :
				\begin{itemize}
					\item {\scriptsize une pile $S$}
					\item {\scriptsize un environnement $E$}
					\item {\scriptsize une chaîne de contrôle $C$}
					\item {\scriptsize un dépôt $D$}
				\end{itemize} 
			\end{itemize}
			\end{column}
		\end{columns}
		\
	
		\begin{block}{Règles de la machine SECD}
		\begin{enumerate}
			\item  {\scriptsize $\langle S,E,b~C,D\rangle 
				\longmapsto_{secd}
				\langle b~S,E,C,D\rangle$}
			
			\item  {\scriptsize $\langle S,E,X$ $C,D\rangle 
				\longmapsto_{secd}
				\langle V$ $S,E,C,D\rangle$ où $V = E(X)$}
			
			\item  {\scriptsize $\langle S,E,\langle X,C'\rangle$ $C,D\rangle \longmapsto_{secd}
				\langle\langle\langle X,C'\rangle,E\rangle$ $S,E,C,D\rangle$}
			
			\item  {\scriptsize $\langle V$ $\langle\langle X,C'\rangle,E'\rangle$ $S,E,ap$ $C,D\rangle 
				\longmapsto_{secd}
				\langle\epsilon,E'[X \leftarrow V],C',\langle S,E,C,D\rangle\rangle$}
			
			\item  {\scriptsize $\langle V$ $S,E,\emptyset,\langle S',E',C',D\rangle\rangle \longmapsto_{secd}
				\langle V$ $S',E',C',D\rangle$}
			
			\item  {\scriptsize $\langle b_{1}$ $...$ $b_{n}~S,E,prim_{o^{n}}$ $C,D\rangle \longmapsto_{secd}
				\langle V$ $S,E,C,D\rangle$ où $V = \delta(o^{n},b_1,...b_{n})$}
		\end{enumerate}
		\end{block}	
		 
	\end{frame}

	\section{Programmation réactive synchrone concurrente}
	
	\begin{frame}
		\frametitle{Programmation réactive synchrone concurrente}
		
		\subsection{Concept}
			\begin{block}{Programmation réactive}
				\textbf{La programmation réactive} est un paradigme de programmation visant à conserver une cohérence d'ensemble en propageant les modifications d'une source réactive aux éléments dépendants de cette source. 
			\end{block}
			\bigbreak
			
					
			\begin{block}{Synchrone}
				\textbf{Synchrone} signifie que les informations seront obtenues de manière immédiate. 
			\end{block}
			\bigbreak
			
			
			\begin{block}{Concurrente}
				\textbf{Concurrente} signifie que plusieurs processus vont se dérouler durant le même instant logique. 
			\end{block}
			
			
	\end{frame}

	\begin{frame}
		\subsection{Sémantique de la machine}
		\frametitle{Machines TTS : Sémantique de la machine }
		\begin{itemize}
			\item[]\textbf{\small Création d'un thread :} \\
			{\scriptsize$\langle\langle\langle\langle X,C'\rangle, E\rangle~S,E,spawn~C,D\rangle,TL,SI\rangle 
				\longrightarrow_{TTS} 
				\langle\langle S,E,C,D\rangle,TL~\langle S,E,C',D\rangle,SI\rangle$}
			\item[]
			
			\item[] \textbf{\small Initialisation d'un signal :} \\
			{\scriptsize$\langle\langle S,E,init~C,D\rangle,TL,SI\rangle 
				\longrightarrow_{TTS}
				\langle\langle s~S,E,C,D\rangle,TL,SI'\rangle$
				\\avec $\iota(SI) = (s,SI')$}
			\item[]
			
			\item[] \textbf{\small Émettre :} \\
			{\scriptsize $\langle\langle s~S,E,emit~C,D\rangle,TL,SI\rangle 
				\longrightarrow_{TTS}
				\langle\langle S,E,C,D\rangle,TL~ST,SI'\rangle$
				\\ avec $\varepsilon(s,SI) = (ST,SI)'$}
			\item[]
			
			\item[] \textbf{\small Récupération de thread :}\\
			{\scriptsize $\langle\langle S,E,\epsilon,\emptyset\rangle,\langle S',E',C,D\rangle~TL,SI\rangle 
				\longrightarrow_{TTS} 
				\langle\langle S',E',C,D\rangle,TL,SI\rangle$}
			\item[]
			
			\item[] \textbf{\small Fin d'un instant logique :}\\
			{\scriptsize $\langle\langle S,E,\epsilon,\emptyset \rangle ,\emptyset,SI\rangle 
				\longrightarrow_{TTS} 
				\langle\langle S,E,\epsilon,\emptyset\rangle,TL,SI'\rangle$
				avec $\tau(SI) = (TL,SI')$ }
		\end{itemize}	
	\end{frame}

	\begin{frame}
	\subsection{Sémantique de la machine}
	\frametitle{Machines TTS : Sémantique de la machine }
		\begin{itemize}
			\item[] \textbf{\small Test d'un signal présent :} \\
			{\scriptsize$\langle\langle \langle\langle X',C''\rangle,E\rangle~\langle\langle X,C'\rangle,E\rangle~s~S,E,present~C,D\rangle,TL,SI\rangle 
				\longrightarrow_{TTS} 
				\langle\langle S,E,C'~C,D\rangle,TL,SI\rangle$ 
				avec $SI(s) = \langle vraie,ST\rangle$}
			\item[]
			
			\item[] \textbf{\small Test d'un signal non présent avec thread remplaçable :}\\
				{\scriptsize$\langle\langle \langle\langle X',C''\rangle,E\rangle~\langle\langle X,C'\rangle,E\rangle~s~S,E,present~C,D\rangle,\langle S',E',C''',D'\rangle$ $TL,SI\rangle 
				\longrightarrow_{TTS} 
				\langle\langle S',E',C''',D'\rangle,TL,SI'\rangle~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~$ 
				avec $SI(s) = \langle faux,ST\rangle~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~$
				et $SI'(s) = \langle faux,ST~\langle \langle\langle X',C''\rangle,E\rangle~\langle\langle X,C'\rangle,E\rangle~s~S,E,present~C,D\rangle\rangle$}
			\item[]
			
			\item[] \textbf{\small Test d'un signal non présent sans thread remplaçable :}\\
				{\scriptsize $\langle\langle \langle\langle X',C''\rangle,E\rangle~\langle\langle X,C'\rangle,E\rangle~s~S,E,C,D\rangle,\emptyset,SI\rangle 
					\longrightarrow_{TTS} 
					\langle\langle \emptyset,\epsilon,\emptyset,\emptyset\rangle,\emptyset,SI'\rangle~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~$
					avec $SI(s) = \langle faux,ST\rangle~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~$
					et $SI'(s) = \langle faux,ST~\langle\langle\langle X',C''\rangle,E\rangle~\langle\langle X,C'\rangle,E\rangle~s~S,E,present~C,D\rangle\rangle$}
		\end{itemize}	
	\end{frame}
